\setbeamerfont{footline}{parent=frametitle}

% Macro to hold the “Break/End” footer text
\newcommand{\sessionbreaktext}{}

% Redefine the footline to use that text centered at the bottom
\setbeamertemplate{footline}{%
  \begin{beamercolorbox}[wd=\paperwidth,ht=2.5ex,dp=1ex,center]{author in head/foot}%
    \usebeamerfont{footline}\sessionbreaktext%
  \end{beamercolorbox}%
}

% sessionframe takes two args: (1) the frame title, (2) the break/end text
\newenvironment{sessionframe}[2]{%
  \renewcommand{\sessionbreaktext}{#2}%
  \begin{frame}[t]{#1}%
}{%
  \end{frame}%
}
